1//! Storage profiles — presets for different use cases.
2//!
3//! Each profile specifies which domains and sources to include,
4//! with a target storage budget.
5
6use crate::domain::Domain;
7use serde::{Deserialize, Serialize};
8
9/// A storage profile — a curated selection of knowledge for a use case.
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct Profile {
12    /// Profile identifier (e.g. "survival", "developer").
13    pub id: String,
14    /// Human-readable name.
15    pub name: String,
16    /// Description of the use case.
17    pub description: String,
18    /// Target storage budget in megabytes.
19    pub budget_mb: u64,
20    /// Domains to include.
21    pub domains: Vec<Domain>,
22    /// Whether to include external sources (Wikipedia, etc.).
23    pub include_external: bool,
24}
25
26impl Profile {
27    /// Survival kit — essential survival, medical, and practical knowledge.
28    #[must_use]
29    pub fn survival() -> Self {
30        Self {
31            id: "survival".into(),
32            name: "Survival Kit".into(),
33            description: "Essential survival, medical, and practical knowledge. Fits on a phone."
34                .into(),
35            budget_mb: 2_000,
36            domains: vec![
37                Domain::Medicine,
38                Domain::Survival,
39                Domain::Agriculture,
40                Domain::Construction,
41                Domain::Repair,
42                Domain::Communication,
43            ],
44            include_external: true,
45        }
46    }
47
48    /// Homesteader — self-sufficiency knowledge.
49    #[must_use]
50    pub fn homesteader() -> Self {
51        Self {
52            id: "homesteader".into(),
53            name: "Homesteader".into(),
54            description:
55                "Self-sufficiency: farming, building, repair, medicine, plus science fundamentals."
56                    .into(),
57            budget_mb: 5_000,
58            domains: vec![
59                Domain::Medicine,
60                Domain::Survival,
61                Domain::Agriculture,
62                Domain::Construction,
63                Domain::Repair,
64                Domain::Communication,
65                Domain::Chemistry,
66                Domain::Biology,
67                Domain::EarthScience,
68                Domain::Mathematics,
69            ],
70            include_external: true,
71        }
72    }
73
74    /// Developer — programming, systems, math.
75    #[must_use]
76    pub fn developer() -> Self {
77        Self {
78            id: "developer".into(),
79            name: "Developer".into(),
80            description: "Programming references, algorithms, systems knowledge, math.".into(),
81            budget_mb: 3_000,
82            domains: vec![
83                Domain::Computing,
84                Domain::Mathematics,
85                Domain::Statistics,
86                Domain::Physics,
87            ],
88            include_external: false,
89        }
90    }
91
92    /// Educator — broad coverage for teaching.
93    #[must_use]
94    pub fn educator() -> Self {
95        Self {
96            id: "educator".into(),
97            name: "Educator".into(),
98            description: "Broad knowledge base for teaching and learning across all domains."
99                .into(),
100            budget_mb: 8_000,
101            domains: Domain::all().to_vec(),
102            include_external: true,
103        }
104    }
105
106    /// Full — everything.
107    #[must_use]
108    pub fn full() -> Self {
109        Self {
110            id: "full".into(),
111            name: "Complete Knowledge Base".into(),
112            description: "All domains, all sources. The full foundation of knowing.".into(),
113            budget_mb: 10_000,
114            domains: Domain::all().to_vec(),
115            include_external: true,
116        }
117    }
118
119    /// All built-in profiles.
120    #[must_use]
121    pub fn all_profiles() -> Vec<Self> {
122        vec![
123            Self::survival(),
124            Self::homesteader(),
125            Self::developer(),
126            Self::educator(),
127            Self::full(),
128        ]
129    }
130}

132#[cfg(test)]
133mod tests {
134    use super::*;
135
136    #[test]
137    fn profile_survival() {
138        let p = Profile::survival();
139        assert_eq!(p.id, "survival");
140        assert!(p.budget_mb <= 2_000);
141        assert!(p.domains.contains(&Domain::Medicine));
142    }
143
144    #[test]
145    fn profile_developer() {
146        let p = Profile::developer();
147        assert!(p.domains.contains(&Domain::Computing));
148        assert!(!p.include_external);
149    }
150
151    #[test]
152    fn profile_full_has_all_domains() {
153        let p = Profile::full();
154        assert_eq!(p.domains.len(), Domain::all().len());
155    }
156
157    #[test]
158    fn all_profiles() {
159        let profiles = Profile::all_profiles();
160        assert_eq!(profiles.len(), 5);
161    }
162
163    #[test]
164    fn profile_serde_roundtrip() {
165        let p = Profile::survival();
166        let json = serde_json::to_string(&p).unwrap();
167        let decoded: Profile = serde_json::from_str(&json).unwrap();
168        assert_eq!(p.id, decoded.id);
169        assert_eq!(p.budget_mb, decoded.budget_mb);
170    }
171}
