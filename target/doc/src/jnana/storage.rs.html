1//! Storage budget calculator.
2//!
3//! Given available disk space and a profile, determines what fits.
4
5use crate::profile::Profile;
6use crate::source::Source;
7use serde::{Deserialize, Serialize};
8
9/// Storage budget analysis.
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct StorageBudget {
12    /// Available space in megabytes.
13    pub available_mb: u64,
14    /// Profile being evaluated.
15    pub profile_id: String,
16    /// Internal knowledge size (AGNOS crate data) in megabytes.
17    pub internal_mb: u64,
18    /// External sources that fit within budget.
19    pub fits: Vec<String>,
20    /// External sources that don't fit.
21    pub excluded: Vec<String>,
22    /// Total size of selected content in megabytes.
23    pub total_mb: u64,
24    /// Remaining space in megabytes.
25    pub remaining_mb: u64,
26}
27
28/// Calculate what fits on the given storage for a profile.
29#[must_use]
30pub fn calculate(available_mb: u64, profile: &Profile, sources: &[Source]) -> StorageBudget {
31    // Internal knowledge is tiny — estimate 100MB for all AGNOS crate data
32    let internal_mb = 100;
33    let mut remaining = available_mb.saturating_sub(internal_mb);
34    let mut fits = Vec::new();
35    let mut excluded = Vec::new();
36    let mut total = internal_mb;
37
38    // Sort sources by size (smallest first) to maximize count
39    let mut sorted: Vec<&Source> = sources
40        .iter()
41        .filter(|s| s.enabled && profile.domains.contains(&s.domain))
42        .collect();
43    sorted.sort_by_key(|s| s.size_mb);
44
45    for source in sorted {
46        if source.size_mb <= remaining {
47            remaining -= source.size_mb;
48            total += source.size_mb;
49            fits.push(source.name.clone());
50        } else {
51            excluded.push(source.name.clone());
52        }
53    }
54
55    StorageBudget {
56        available_mb,
57        profile_id: profile.id.clone(),
58        internal_mb,
59        fits,
60        excluded,
61        total_mb: total,
62        remaining_mb: remaining,
63    }
64}

66#[cfg(test)]
67mod tests {
68    use super::*;
69    use crate::domain::Domain;
70    use crate::source::SourceKind;
71
72    #[test]
73    fn budget_empty() {
74        let profile = Profile::survival();
75        let budget = calculate(5000, &profile, &[]);
76        assert_eq!(budget.internal_mb, 100);
77        assert_eq!(budget.total_mb, 100);
78        assert!(budget.fits.is_empty());
79    }
80
81    #[test]
82    fn budget_fits_small_sources() {
83        let profile = Profile::survival();
84        let sources = vec![Source {
85            id: "med".into(),
86            name: "Medical Encyclopedia".into(),
87            domain: Domain::Medicine,
88            kind: SourceKind::Zim,
89            url: "".into(),
90            size_mb: 1500,
91            checksum: None,
92            enabled: true,
93            notes: "".into(),
94        }];
95        let budget = calculate(5000, &profile, &sources);
96        assert_eq!(budget.fits.len(), 1);
97        assert!(budget.excluded.is_empty());
98    }
99
100    #[test]
101    fn budget_excludes_too_large() {
102        let profile = Profile::survival();
103        let sources = vec![Source {
104            id: "wiki".into(),
105            name: "Full Wikipedia".into(),
106            domain: Domain::Medicine,
107            kind: SourceKind::Zim,
108            url: "".into(),
109            size_mb: 50_000,
110            checksum: None,
111            enabled: true,
112            notes: "".into(),
113        }];
114        let budget = calculate(5000, &profile, &sources);
115        assert!(budget.fits.is_empty());
116        assert_eq!(budget.excluded.len(), 1);
117    }
118}
