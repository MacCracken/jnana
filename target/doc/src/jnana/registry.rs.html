1//! In-memory knowledge registry.
2
3use crate::domain::Domain;
4use crate::entry::Entry;
5use crate::error::{JnanaError, Result};
6use std::collections::HashMap;
7
8/// The knowledge registry — holds all entries in memory.
9pub struct Registry {
10    entries: HashMap<String, Entry>,
11}
12
13impl Registry {
14    /// Create an empty registry.
15    #[must_use]
16    pub fn new() -> Self {
17        Self {
18            entries: HashMap::new(),
19        }
20    }
21
22    /// Register an entry. Overwrites if the ID already exists.
23    pub fn register(&mut self, entry: Entry) {
24        tracing::debug!(id = %entry.id, domain = %entry.domain, "registered entry");
25        self.entries.insert(entry.id.clone(), entry);
26    }
27
28    /// Get an entry by ID.
29    #[must_use]
30    pub fn get(&self, id: &str) -> Option<&Entry> {
31        self.entries.get(id)
32    }
33
34    /// Get an entry by ID, returning an error if not found.
35    pub fn get_or_err(&self, id: &str) -> Result<&Entry> {
36        self.entries
37            .get(id)
38            .ok_or_else(|| JnanaError::EntryNotFound(id.into()))
39    }
40
41    /// List all entry IDs, sorted.
42    #[must_use]
43    pub fn list_ids(&self) -> Vec<&str> {
44        let mut ids: Vec<&str> = self.entries.keys().map(|s| s.as_str()).collect();
45        ids.sort_unstable();
46        ids
47    }
48
49    /// List all entries, sorted by ID.
50    #[must_use]
51    pub fn list(&self) -> Vec<&Entry> {
52        let mut entries: Vec<&Entry> = self.entries.values().collect();
53        entries.sort_by(|a, b| a.id.cmp(&b.id));
54        entries
55    }
56
57    /// Filter entries by domain.
58    #[must_use]
59    pub fn by_domain(&self, domain: Domain) -> Vec<&Entry> {
60        self.entries
61            .values()
62            .filter(|e| e.domain == domain)
63            .collect()
64    }
65
66    /// Number of entries.
67    #[must_use]
68    #[inline]
69    pub fn len(&self) -> usize {
70        self.entries.len()
71    }
72
73    #[must_use]
74    #[inline]
75    pub fn is_empty(&self) -> bool {
76        self.entries.is_empty()
77    }
78
79    /// Total estimated size of all entries in bytes.
80    #[must_use]
81    pub fn total_size(&self) -> usize {
82        self.entries.values().map(|e| e.estimated_size()).sum()
83    }
84
85    /// Count entries per domain.
86    #[must_use]
87    pub fn domain_counts(&self) -> HashMap<Domain, usize> {
88        let mut counts = HashMap::new();
89        for entry in self.entries.values() {
90            *counts.entry(entry.domain).or_insert(0) += 1;
91        }
92        counts
93    }
94}
95
96impl Default for Registry {
97    fn default() -> Self {
98        Self::new()
99    }
100}

102#[cfg(test)]
103mod tests {
104    use super::*;
105    use crate::entry::{Constant, EntryKind};
106
107    fn make_entry(id: &str, domain: Domain) -> Entry {
108        Entry {
109            id: id.into(),
110            title: id.into(),
111            domain,
112            summary: format!("Test entry: {id}"),
113            kind: EntryKind::Constant(Constant {
114                symbol: "x".into(),
115                value: "42".into(),
116                unit: "".into(),
117                numeric: 42.0,
118                uncertainty: None,
119                authority: "test".into(),
120            }),
121            source: "test".into(),
122            tags: vec![],
123        }
124    }
125
126    #[test]
127    fn registry_empty() {
128        let reg = Registry::new();
129        assert!(reg.is_empty());
130        assert_eq!(reg.len(), 0);
131    }
132
133    #[test]
134    fn registry_register_and_get() {
135        let mut reg = Registry::new();
136        reg.register(make_entry("pi", Domain::Mathematics));
137        assert_eq!(reg.len(), 1);
138        assert!(reg.get("pi").is_some());
139        assert!(reg.get("missing").is_none());
140    }
141
142    #[test]
143    fn registry_by_domain() {
144        let mut reg = Registry::new();
145        reg.register(make_entry("pi", Domain::Mathematics));
146        reg.register(make_entry("c", Domain::Physics));
147        reg.register(make_entry("e", Domain::Mathematics));
148        let math = reg.by_domain(Domain::Mathematics);
149        assert_eq!(math.len(), 2);
150    }
151
152    #[test]
153    fn registry_domain_counts() {
154        let mut reg = Registry::new();
155        reg.register(make_entry("a", Domain::Mathematics));
156        reg.register(make_entry("b", Domain::Mathematics));
157        reg.register(make_entry("c", Domain::Physics));
158        let counts = reg.domain_counts();
159        assert_eq!(counts[&Domain::Mathematics], 2);
160        assert_eq!(counts[&Domain::Physics], 1);
161    }
162
163    #[test]
164    fn registry_total_size() {
165        let mut reg = Registry::new();
166        reg.register(make_entry("a", Domain::Mathematics));
167        assert!(reg.total_size() > 0);
168    }
169}
